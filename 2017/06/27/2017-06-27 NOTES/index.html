#rpm#yum
Jun 27, 2017

更换国内yum

cd /etc/yum.reposd.d/（进入仓库目）

curl -O http://mirrors.163.com/.help/CentOS7-Base-163.repo（下载163源）

yum clean all(清除yum缓存)

yum makecache(生成yum缓存)

yum install -y epel-release（安装扩展源）

yum list |grep epel（查询扩展源）

yum install -y vim --downloadonly （yum 下载rpm包）

/var/cache/yum/x86_64/7/*(下载后的路径)

yum install -y vim --downloadonly --downloaddir=/tmp/(指定yum包安装路径)

源码包安装

echo $? 如果返回0就是正常
